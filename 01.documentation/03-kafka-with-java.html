  • Code / Config changes
    • Producer: KafkaProducerDemo.java
      • imports
        • import org.apache.kafka.clients.producer.KafkaProducer;
        • import org.apache.kafka.clients.producer.ProducerRecord;
        • import org.apache.kafka.common.serialization.StringSerializer;
      • Producer class to publish data to topics.
            public class KafkaProducerDemo {

      private static final Logger logger = LoggerFactory.getLogger(KafkaProducerDemo.class);

      public static void main(String[] args) {
          logger.info("Hello World.");

          // Create Producer properties
          final Properties properties = new Properties();
          properties.setProperty("bootstrap.servers", "[::1]:9092");
          properties.setProperty("key.serializer", StringSerializer.class.getName());
          properties.setProperty("value.serializer", StringSerializer.class.getName());

          // Create Producer
          KafkaProducer<String, String> producer = new KafkaProducer<>(properties);

          // Create producer record
          ProducerRecord<String, String> record = new ProducerRecord<>("demo_java", "hello-world 1");
          ProducerRecord<String, String> record2 = new ProducerRecord<>("demo_java", "hello-world 2");

          // Send data
          producer.send(record);
          producer.send(record2);

          // flush and close the producer
          producer.flush();
          producer.close();

          logger.info("completed API execution.");
      }
  }
